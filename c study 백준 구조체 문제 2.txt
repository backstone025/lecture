#include <stdio.h>

typedef struct member{

    int age;
    char name[100];
    int ranking;

}member;



int main (){
    int n;

    scanf("%d", &n);
member x[n];
    for(int i = 0; i < n; i++)
        scanf("%d %s", &(x[i].age), &(x[i].name));




    for (int i = 0; i < n; i++){
        int k = 0;
        for(int j = 0; j < n; j++){

            if(x[i].age > x[j].age)
                k++;

        }

        x[i].ranking = k;

    }




    for(int i = 0; i < n; i++){

        for(int j = 0; j < n; j++){

            if(x[j].ranking == i)
                printf("%d %s\n", x[j].age, x[j].name);
        }

    }





    return 0;
}
